fn mean(&self) -> T

Returns the mean of the Gaussian distribution.

Examples

use rustml::gaussian::*;
use rustml::Normalization;

let a = vec![1.0f32, 2.0, 4.0, 3.0, 6.0, 5.0];
let g = a.gaussian(Normalization::N);
assert_eq!(g.mean(), 3.5);
